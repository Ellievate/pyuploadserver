const fileInput = document.getElementById('file-input');
const dropZone = document.querySelector('.drop-zone');
const progressBar = document.getElementById('progress-bar');
const progressText = document.getElementById('progress-text');

// Add event listeners for drag and drop functionality
dropZone.addEventListener('dragover', (e) => {
    e.preventDefault();
});

dropZone.addEventListener('drop', (e) => {
    e.preventDefault();
    const files = e.dataTransfer.files;
    handleFiles(files);
});

// Add event listener for file input change
fileInput.addEventListener('change', () => {
    const files = fileInput.files;
    handleFiles(files);
});

function handleFiles(files) {
    // Create a new form data object to send the files
    const formData = new FormData();
    for (const file of files) {
        formData.append('files[]', file);
    }

    // Send the files to the server using AJAX
    fetch('/upload', {
        method: 'POST',
        body: formData,
    })
        .then((response) => response.json())
        .then((data) => console.log(data))
        .catch((error) => console.error(error));

    // Update the progress bar and text
    const totalFiles = files.length;
    let uploadedFiles = 0;

    function updateProgress() {
        const progress = (uploadedFiles / totalFiles) * 100;
        progressBar.value = progress;
        progressText.textContent = `${Math.round(progress)}%`;
    }

    // Simulate file upload progress
    setInterval(() => {
        if (uploadedFiles < totalFiles) {
            uploadedFiles++;
            updateProgress();
        }
    }, 500);
}

from flask import Flask, request, send_file
from werkzeug.utils import secure_filename
from cryptography.fernet import Fernet
import os

app = Flask(__name__)

# Set encryption password
password = "my_secret_password"
key = Fernet.generate_key()
cipher_suite = Fernet(key)

# Create upload folder if it doesn't exist
upload_folder = "uploads"
if not os.path.exists(upload_folder):

@app.route("/", methods=["GET", "POST"])

    if request.method == "POST":
        file = request.files["file"]
        filename = secure_filename(file.filename)
        filepath = os.path.join(upload_folder, filename)
        file.save(filepath)

        # Encrypt the file
        with open(filepath, "rb") as f:
            encrypted_data = cipher_suite.encrypt(f.read())
        with open(filepath, "wb") as f:
            f.write(encrypted_data)

        return "File uploaded and encrypted successfully!"
    return send_file("index.html")
